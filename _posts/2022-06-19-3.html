part1, part2 = nums[::2], nums[1::2]

parts_sum = [{0}, {0}]
for i, numbers in enumerate([part1, part2]):
    for num in numbers:
        parts_sum[i] |= {num + s for s in parts_sum[i]}
part1_sum, part2_sum = parts_sum
part2_sum = sorted(part2_sum)
